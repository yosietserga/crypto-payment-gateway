Payout API Endpoints

POST /payouts

Create Payout

Creates a new cryptocurrency payout to a recipient address.

Request Parameters

Parameter Type Required Description
amount Numeric Yes Amount to send
recipientAddress String Yes Recipient's crypto wallet address
currency String Yes Cryptocurrency code (e.g. USDT)
network String Yes Blockchain network (e.g. BSC)
webhookUrl String No URL for payout status notifications
metadata Object No Additional data for the payout

Example Requests

cURL
curl -X POST "https://eoscryptopago.com/api/v1/payouts" \
-H "Authorization: Bearer YOUR_API_KEY" \
-H "Content-Type: application/json" \
-d '{
  "amount": 50.00,
  "recipientAddress": "0xA1B2C3D4E5F67890ABCD1234567890ABCDEF1234",
  "currency": "USDT",
  "network": "BSC",
  "webhookUrl": "https://your-website.com/webhook",
  "metadata": {
    "description": "Affiliate payout",
    "reference": "PAYOUT-12345",
    "memo": "May earnings"
  }
}'
JavaScript

PHP
function createPayout() {
    $apiKey = 'YOUR_API_KEY';
    $url = 'https://eoscryptopago.com/api/v1/payouts';
    
    $data = [
        'amount' => 50.00,
        'recipientAddress' => '0xA1B2C3D4E5F67890ABCD1234567890ABCDEF1234',
        'currency' => 'USDT',
        'network' => 'BSC',
        'webhookUrl' => 'https://your-website.com/webhook',
        'metadata' => [
            'description' => 'Affiliate payout',
            'reference' => 'PAYOUT-12345',
            'memo' => 'May earnings'
        ]
    ];
    
    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_POST, true);
    curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($data));
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        'Authorization: Bearer ' . $apiKey,
        'Content-Type: application/json'
    ]);
    
    $response = curl_exec($ch);
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    if ($httpCode == 201) {
        return json_decode($response, true);
    } else {
        $error = json_decode($response, true);
        throw new Exception('Error creating payout: ' . 
                           ($error['message'] ?? 'Unknown error'));
    }
}

GET /payouts

List Payouts

Retrieves a list of payouts with optional filtering.

Query Parameters

Parameter Type Required Description
page Integer No Page number (default: 1)
limit Integer No Items per page (default: 10, max: 100)
status String No Filter by status (PENDING, PROCESSING, COMPLETED, FAILED)
currency String No Filter by cryptocurrency
fromDate String No ISO8601 start date
toDate String No ISO8601 end date

Example Requests

cURL
curl -X GET "https://eoscryptopago.com/api/v1/payouts?page=1&limit=10&status=COMPLETED" \
-H "Authorization: Bearer YOUR_API_KEY"
JavaScript
const getPayouts = async (params = {}) => {
  try {
    const queryParams = new URLSearchParams({
      page: params.page || 1,
      limit: params.limit || 10,
      status: params.status || '',
      currency: params.currency || '',
      fromDate: params.fromDate || '',
      toDate: params.toDate || ''
    }).toString();
    
    const response = await fetch(`https://eoscryptopago.com/api/v1/payouts?${queryParams}`, {
      method: 'GET',
      headers: {
        'Authorization': 'Bearer YOUR_API_KEY'
      }
    });
    
    const data = await response.json();
    console.log('Payouts:', data);
    return data;
  } catch (error) {
    console.error('Error fetching payouts:', error);
    throw error;
  }
};
PHP
function getPayouts($params = []) {
    $apiKey = 'YOUR_API_KEY';
    $queryParams = http_build_query([
        'page' => $params['page'] ?? 1,
        'limit' => $params['limit'] ?? 10,
        'status' => $params['status'] ?? '',
        'currency' => $params['currency'] ?? '',
        'fromDate' => $params['fromDate'] ?? '',
        'toDate' => $params['toDate'] ?? ''
    ]);
    
    $url = 'https://eoscryptopago.com/api/v1/payouts?' . $queryParams;
    
    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        'Authorization: Bearer ' . $apiKey
    ]);
    
    $response = curl_exec($ch);
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    if ($httpCode == 200) {
        return json_decode($response, true);
    } else {
        $error = json_decode($response, true);
        throw new Exception('Error fetching payouts: ' . 
                           ($error['message'] ?? 'Unknown error'));
    }
}
